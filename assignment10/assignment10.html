  • Generate a line plot with the "usd_to_jpy" series, label the axes and the title

exchange_rates["bool_cond"] = exchange_rates["usd_to_jpy"] < 100

# line plot
plt.plot("date", "usd_to_jpy", data = exchange_rates)
plt.xlabel("Time")
plt.ylabel("Rate")
plt.title("Exchange Rate: USD & JPY")

vec_y = exchange_rates["usd_to_jpy"]
plt.fill_between(x= exchange_rates["date"],
                 y1 = vec_y.min(),
                 y2 = vec_y.max(),
                 where = exchange_rates["bool_cond"],
                 alpha = 0.2,color = "red")

plt.show()

month_config = pd.Grouper(key='date', freq='m')

(exchange_rates.groupby(month_config)
              .agg(std_usd_to_rmb = ("usd_to_rmb","std"))
              .reset_index())
